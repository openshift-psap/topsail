        kube_pod_resource_request{resource="memory"}

        )
      )

        kube_pod_resource_request{resource="cpu"}
        *
        on(node) group_left(role) (
          max by (node) (kube_node_role{role=~".+"})
        )
      )

        (
          1 - rate(node_cpu_seconds_total{mode="idle"}[2m])
          *

        )
        *
        on(node) group_left(role) (
          max by (node) (kube_node_role{role=~".+"})
        )
      )
